import React, { useState } from "react";
import { useClickOutside } from "@mantine/hooks";

import Button from "@/components/Button";

export default function UseClickOutside() {
  const [isOpen, setIsOpen] = useState(false);

  const ref = useClickOutside(() => setIsOpen(false));

  return (
    <div className="flex justify-center" ref={ref}>
      <Button onClick={() => setIsOpen(true)}>Open</Button>

      {isOpen && (
        <div className="absolute w-full max-w-xl translate-y-12 rounded-md border-wrapper p-4">
          <h2 className="text-lg font-medium">Lorem Ipsum</h2>
          <p className="text-muted">
            Lorem ipsum dolor, sit amet consectetur adipisicing elit. Nihil
            incidunt sequi suscipit placeat dolorum doloribus inventore minima
            excepturi dolore sit error quasi, tenetur autem harum impedit.
          </p>
        </div>
      )}
    </div>
  );
}

import React from "react";
import { useFullscreen } from "@mantine/hooks";

import Button from "@/components/Button";

export default function UseFullscreen() {
  const { toggle, fullscreen } = useFullscreen();

  return (
    <Button className={`${fullscreen && "bg-red-600"}`} onClick={toggle}>
      {fullscreen ? "Exit" : "Enter"} Fullscreen
    </Button>
  );
}

```tsx title="[name].tsx" showLineNumbers {2,5,13}#add
import React from "react";
import { useDebouncedState } from "@mantine/hooks";

export default function UseDebouncedState() {
  const [value, setValue] = useDebouncedState("", 200);

  return (
    <div>
      <input
        type="text"
        placeholder="type..."
        className="rounded-md border-wrapper p-2 outline-none"
        onChange={(event) => setValue(event.currentTarget.value)}
      />

      <p className="mt-4">Debounced value: {value}</p>
    </div>
  );
}
```
