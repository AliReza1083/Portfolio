```tsx title="[name].tsx" showLineNumbers
import React from "react";
import {
  PDFDownloadLink,
  Document,
  Page,
  Text,
  View,
  StyleSheet,
  Font,
} from "@react-pdf/renderer";

import Button from "@/components/Button";

Font.register({
  family: "Open Sans",
  fonts: [
    {
      src: "https://cdn.jsdelivr.net/npm/open-sans-all@0.1.3/fonts/open-sans-regular.ttf",
    },
    {
      src: "https://cdn.jsdelivr.net/npm/open-sans-all@0.1.3/fonts/open-sans-600.ttf",
      fontWeight: 600,
    },
  ],
});

// Define the styles
const styles = StyleSheet.create({
  page: {
    fontFamily: "Open Sans",
    padding: "30",
    backgroundColor: "#635BFF",
    color: "white",
    display: "flex",
    justifyContent: "center",
    alignItems: "center",
  },
  businessName: {
    width: 300,
    marginBottom: 20,
    fontSize: 12,
  },
  section: {
    backgroundColor: "white",
    color: "#7A7A7B",
    width: 300,
    borderRadius: 8,
    padding: 16,
    marginBottom: 16,
  },
  flexRow: {
    flexDirection: "row",
    justifyContent: "space-between",
    alignItems: "center",
    gap: 8,
  },
  wrapperText: {
    display: "flex",
    flexDirection: "column",
    gap: 2,
    borderBottom: "1px solid #E8E9EC",
    flexGrow: 1,
    paddingBottom: 16,
  },
  mutedText: {
    fontSize: 12,
  },
  price: {
    fontSize: 26,
    color: "black",
    fontWeight: "semibold",
    letterSpacing: -0.2,
  },
  bottomInfo: {
    gap: 4,
    marginTop: 16,
  },
  value: {
    fontSize: 12,
    color: "black",
  },
});

// Invoice Document Component
const InvoiceDocument = () => (
  <Document>
    <Page size="A5" style={styles.page}>
      <Text style={styles.businessName}>Stripe Shop</Text>
      <View style={styles.section}>
        <View style={styles.flexRow}>
          <View style={styles.wrapperText}>
            <Text style={styles.mutedText}>Receipt from X</Text>
            <Text style={styles.price}>$8.00</Text>
            <Text style={styles.mutedText}>Paid July 17, 2024</Text>
          </View>
          <Text></Text>
        </View>
        <View style={styles.bottomInfo}>
          <View style={styles.flexRow}>
            <Text style={styles.mutedText}>Receipt number</Text>
            <Text style={styles.value}>2093 2391 1280</Text>
          </View>
          <View style={styles.flexRow}>
            <Text style={styles.mutedText}>Invoice number</Text>
            <Text style={styles.value}>E12E963E-0002</Text>
          </View>
          <View style={styles.flexRow}>
            <Text style={styles.mutedText}>Payment methode</Text>
            <Text style={styles.value}>Visa 1607</Text>
          </View>
        </View>
      </View>
      <View style={styles.section}>
        <Text style={styles.value}>Receipt #1232-3412-1313</Text>
      </View>
    </Page>
  </Document>
);

export default function ReactPDF() {
  return (
    <div>
      <PDFDownloadLink document={<InvoiceDocument />} fileName="invoice.pdf">
        {({ loading }) => (
          <Button>{loading ? "Generating PDF..." : "Generate Invoice"}</Button>
        )}
      </PDFDownloadLink>
    </div>
  );
}
```
