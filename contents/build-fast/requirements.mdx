## cn

It is used for making a reusable component for adding new `Tailwind Utilities`. We will need it.

```bash title="Terminal"
pnpm i clsx tailwind-merge
```

```ts title="utils/index.ts"
import clsx, { type ClassValue } from "clsx";
import { twMerge } from "tailwind-merge";

// this function is for reusable components, e.g buttons
export function cn(...inputs: ClassValue[]): any {
  return twMerge(clsx(inputs));
}
```

"use client";

import React, { type ReactNode } from "react";
import { AppProgressBar as ProgressBar } from "next-nprogress-bar";

const ProgressBarProvider = ({ children }: { children: ReactNode }) => {
  return (
    <>
      {children}
      <ProgressBar
        height="2px"
        color="hsl(var(--primary))"
        options={{ showSpinner: false }}
        shallowRouting
      />
    </>
  );
};

export default ProgressBarProvider;
